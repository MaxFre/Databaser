create table person
( pid           serial primary key,
  fnamn       text not null,
  enamn     text not null
);		


create table kontaktperson
(pid 	integer,
 band 	text not null,

primary key (band),
foreign key (pid) references person(pid)
);


create table bandinfo 
(pid 	integer,
 band 	text not null, 
land 	text not null,

primary key (pid),
foreign key (pid) references person(pid)
);

create table jobbare
(pid 	       integer,
personnummer   integer, 
adress         text not null,

primary key (pid),
foreign key (pid) references person(pid)
);


create table spelschema
(band      text not null,
scene      text not null,
tid        text not null,
land       text not null,
 
primary key(scene,tid), 
foreign key (band) references bandinfo(band)
);
